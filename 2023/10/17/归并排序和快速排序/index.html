    if (l == h) return new int[]{nums[l]};
    //求中间值,
    int mid = l + (h - l) / 2;
    // 注意此时的l和h的改变，该步骤相当于将数组分开的过程
    //递归排序左数组，最先mid一直迭代，直到mid=0

    //递归右排序
    int[] rightArr = myMergeSort(nums, mid + 1, h);

    public static void myQuickSort(int[] nums, int low, int high) {
    int i, j, temp, t;
    //保证数组下表左边小于右边，确保low所在位置的元素小于high所在位置的元素
    // 保证主元位置正确
    if (low > high) return;
    i = low;//low为数组低位，i为左边数组位置
    j = high;//high为数组高位，j为右边数组位置
    //temp就是主元
    temp = nums[low];//low为数组第一位
    //分别递归主元左右两边的元素，让左边元素都小于主元，右边元素都大于主元
    //i的最终位置就是主演的位置
    while (i < j) {
    //先看右边，依次往左递减，直到找到小于主元的元素
    //i<j确保i,j位置正确
    while (temp <= nums[j] && i < j) {
    j–;}
    //再看左边，依次往右递增，直到找到大于主元的元素
    while (temp >= nums[i] && i < j) {
    i++;
    }
    //如果满足条件则交换
    // 此判断条件防止i和j在进行上述两步骤的时候，i和j交换浪费时间
    if (i < j) {
    t = nums[j];
    nums[j] = nums[i];
    nums[i] = t;
    }
    }
    //最后将主演为与i相等位置的数字交换，让主元归位
    nums[low] = nums[i];
    nums[i] = temp;
    //递归调用左边数组
    myQuickSort(nums, low, i - 1);
    //递归调用右边数组
    myQuickSort(nums, i + 1, high);
    }
